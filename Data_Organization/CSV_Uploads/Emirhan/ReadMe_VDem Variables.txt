--Identifiers--
country_name - contemporary country names
country_text_id	- abbreviated country names
country_id - country ID as specified by V-Dem
year - year, from 2020 to 2024

--Political Variables - Gender
v2elfemrst - female suffrage, ordinal, 0-no, 1-restricted, 2-universal
v2fsuffrage - percentage of enfranchised female adults, interval
v2pepwrgen** - power distributed by gender, ordinal, 0-4, 4 being most equal
v2clgencl - gender equality with respect to civil liberties, ordinal, 0-4, 4 being most equal
v2peapsgen** - access to public services (including healthcare) by gender, ordinal, 0-4, 4 being most equal
v2xpe_exlgender - exclusion by gender, interval, 0-1, 1 being most excluded from political rights
v2x_gender** - women political empowerment index, interval, 0-1
e_v2x_gender_5C - ordinal variant
v2x_gencl - women civil liberties index, interval, 0-1
e_v2x_gencl_5C - ordinal variant
v2x_gencs - women civil society participation index, interval, 0-1
e_v2x_gencs_5C - ordinal variant
v2x_genpp** - women polticial participation, representation index, interval, 0-1
e_v2x_genpp_5C** - ordinal variant

--Political Variables - Country
v2x_libdem** - liberal democracy index, interval, 0-1, 1 being ideal liberal democracy
e_v2x_libdem_5C** - ordinal variant
v2x_civlib - civil liberties index, interval, 0-1
e_v2x_civlib_5C - ordinal variant
v2x_clphy - physical violence index (by state), interval, 0-1
v2x_clpol - political civil liberties index, interval, 0-1
e_v2x_clpol_5C - ordinal variant
v2x_clpriv - private civil liberties index, interval, 0-1
e_v2x_clpriv_5C - ordinal variant
v2elgvsuflvl - suffrage level, nominal
v2pepwrses - power distributed according to socioeconomic position, ordinal, 0 to 4, 4 being most equal
v2cacamps - political polarization, ordinal, 0-4, 4 being most polarized
v2smpolsoc - polarization of society, ordinal, 0-4, 4 being least polarized
v2x_neopat - neo-patrimonalism index, interval, 0-1
v2xnp_client - clientalism index, interval, 0-1
v2xnp_regcorr** - regime corruption, interval, 0-1
v2x_corr - political corruption index, interval, 0-1
e_v2x_corr_5C - ordinal variant
v2x_rule - rule of law index, interval, 0-1

--Political Variables - Others
v2peapsgeo - access to public services by rural-urban divide, ordinal, 0-5, 4 begin most equal, 5 slight rural-bias
v2xpe_exlecon - exclusion by socioeconomic group, interval, 0-1
v2xpe_exlgeo - exclusion by rural-urban divide, interval, 0-1
v2xpe_exlpol - exclusion by political group, interval, 0-1
v2xpe_exlsocgr - exclusion by social group, interval, 0-1

--Non V-Dem Variables
e_wbgi_gee** - government effectiveness, interval
e_wbgi_pve - political stability
e_peedgini - educational inequality, gini
e_miferrat - fertility rate
e_pechmor - child mortality rate
e_pematmor** - maternal mortality rate
e_wb_pop - population
